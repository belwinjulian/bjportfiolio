          height: '100%',

      >
        <div style={{ fontWeight: 'bold' }}>Belwin Julian</div>
        <div style={{ fontSize: 36, marginTop: 20, color: '#a1a1aa' }}>
          Full Stack Developer
        </div>
      </div>
    ),
    {
      ...size,
    }
  )
}
